jQuery Callbacks
  • 简单示例
  • 进阶,实现订阅/发布 
          
            var callbacks = $.Callbacks()

            function fn1(value) {
              console.log(value)
            }
            function fn2(value) {
              console.log('fn2 says:' + value)
              return false;
            }
            callbacks.add(fn1)
            callbacks.fire('foo!') //foo

            callbacks.add(fn2)
            callbacks.fire('bar!') //bar! fn2 says: bar!
